    components/ClimaActual.tsx#

    import React, { useEffect, useState } from 'react';
import { ClimaInfo } from '../types/Clima';

const ClimaActual: React.FC = () => {
  const [clima, setClima] = useState<ClimaInfo | null>(null);

  useEffect(() => {
    fetch('https://api.example.com/weather')
      .then((res) => res.json())
      .then((data) => {
        setClima(data);
      });
  }, []);

  const obtenerColor = (temp: number) => {
    if (temp < 10) return 'blue';
    if (temp >= 10 && temp <= 20) return 'orange';
    return 'red';
  };

  if (!clima) return <p>Cargando clima...</p>;

  return (
    <div>
      <p><strong>Localidad:</strong> {clima.localidad}</p>
      <p><strong>Fecha:</strong> {clima.fecha}</p>
      <p><strong>Hora:</strong> {clima.hora}</p>
      <p>
        <strong>Temperatura:</strong>{' '}
        <span style={{ color: obtenerColor(clima.temperatura) }}>
          {clima.temperatura}ÂºC
        </span>
      </p>
    </div>
  );
};

export default ClimaActual;
