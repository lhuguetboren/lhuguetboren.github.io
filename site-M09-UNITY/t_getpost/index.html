    Peticiones GET y POST en Unity, manejo de respuestas JSON, y Network Transport**.#

    1. Peticiones GET en Unity#

    Concepto#

    Una petición GET se usa para solicitar información desde un servidor. En Unity, esto se realiza con UnityWebRequest.Get.

    Código Base#

    using UnityEngine;
using UnityEngine.Networking;
using System.Collections;

public class PeticionGET : MonoBehaviour
{
    public void ClickEnviarGet()
    {
        StartCoroutine(EnviarGet());
    }

    IEnumerator EnviarGet()
    {
        UnityWebRequest www = UnityWebRequest.Get("http://localhost/ex_get.php");
        yield return www.SendWebRequest();

        if (www.result != UnityWebRequest.Result.Success)
        {
            Debug.LogError("Error: " + www.error);
        }
        else
        {
            string respuesta = www.downloadHandler.text;
            Debug.Log("Respuesta del servidor: " + respuesta);
        }
    }
}

    Ejemplo Práctico#

    Imagina que tenemos una API de clima y queremos obtener la temperatura actual de una ciudad:

    IEnumerator ObtenerClima(string ciudad)
{
    string url = "https://api.openweathermap.org/data/2.5/weather?q=" + ciudad + "&appid=TU_API_KEY";
    UnityWebRequest www = UnityWebRequest.Get(url);
    yield return www.SendWebRequest();

    if (www.result == UnityWebRequest.Result.Success)
    {
        Debug.Log("Datos del clima: " + www.downloadHandler.text);
    }
    else
    {
        Debug.LogError("Error en la petición: " + www.error);
    }
}

    2. Peticiones POST en Unity#

    Concepto#

    Las peticiones POST envían datos al servidor, normalmente para almacenar información o realizar autenticaciones.

    Código Base#

    using UnityEngine;
using UnityEngine.Networking;
using System.Collections;

public class PeticionPOST : MonoBehaviour
{
    public void ClickEnviarPost()
    {
        StartCoroutine(EnviarPost());
    }

    IEnumerator EnviarPost()
    {
        WWWForm form = new WWWForm();
        form.AddField("usuario", "Juan");
        form.AddField("edad", "25");

        UnityWebRequest www = UnityWebRequest.Post("http://localhost/ex_post.php", form);
        yield return www.SendWebRequest();

        if (www.result != UnityWebRequest.Result.Success)
        {
            Debug.LogError("Error: " + www.error);
        }
        else
        {
            Debug.Log("Respuesta del servidor: " + www.downloadHandler.text);
        }
    }
}

    Ejemplo Práctico#

    En un juego, al registrar a un usuario en el servidor:

    IEnumerator RegistrarUsuario(string nombre, int edad)
{
    WWWForm form = new WWWForm();
    form.AddField("nombre", nombre);
    form.AddField("edad", edad.ToString());

    UnityWebRequest www = UnityWebRequest.Post("http://localhost/registrar.php", form);
    yield return www.SendWebRequest();

    if (www.result == UnityWebRequest.Result.Success)
    {
        Debug.Log("Registro exitoso: " + www.downloadHandler.text);
    }
    else
    {
        Debug.LogError("Error en la petición: " + www.error);
    }
}

    3. Manejo de Respuestas JSON#

    Concepto#

    Cuando el servidor responde con un JSON, podemos convertirlo en un objeto en Unity usando JsonUtility.FromJson.

    Código Base#

    [System.Serializable]
public class InfoJugador
{
    public string nick;
    public int puntos;
}

IEnumerator ObtenerDatosJugador()
{
    UnityWebRequest www = UnityWebRequest.Get("http://localhost/ex_get.php");
    yield return www.SendWebRequest();

    if (www.result == UnityWebRequest.Result.Success)
    {
        string jsonResponse = www.downloadHandler.text;
        InfoJugador jugador = JsonUtility.FromJson<InfoJugador>(jsonResponse);
        Debug.Log("Jugador: " + jugador.nick + " - Puntos: " + jugador.puntos);
    }
}

    Ejemplo con Respuesta JSON en un Array#

    Si la respuesta JSON contiene un array, necesitamos una clase extra para manejarlo: 

    [System.Serializable]
public class ListaJugadores
{
    public InfoJugador[] jugadores;
}

IEnumerator ObtenerListaJugadores()
{
    UnityWebRequest www = UnityWebRequest.Get("http://localhost/ex_get_lista.php");
    yield return www.SendWebRequest();

    if (www.result == UnityWebRequest.Result.Success)
    {
        string jsonResponse = www.downloadHandler.text;
        ListaJugadores lista = JsonUtility.FromJson<ListaJugadores>(jsonResponse);

        foreach (var jugador in lista.jugadores)
        {
            Debug.Log("Jugador: " + jugador.nick + " - Puntos: " + jugador.puntos);
        }
    }
}
