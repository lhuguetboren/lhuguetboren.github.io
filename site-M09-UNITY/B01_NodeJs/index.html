    Instalacion NodeJs+WebSockets

    Instalación de Node.js con WebSocket: Guía paso a paso#

    Esta guía te ayudará a instalar Node.js y crear un servidor WebSocket básico que puedes usar para comunicarte desde Unity u otras aplicaciones.

    1. 🔧 Instalar Node.js#

    1. Ve a la página oficial: https://nodejs.org/
    2. Descarga la versión LTS (Long Term Support) para tu sistema operativo.
    3. Ejecuta el instalador y sigue los pasos por defecto.
    4. Para verificar que se instaló correctamente, abre una terminal (o consola) y escribe:
    node -v
npm -v

    Deberías ver la versión instalada de Node.js y npm (el gestor de paquetes).

    2. 📁 Crear el proyecto Node.js#

    1. Abre una terminal y crea una carpeta para tu proyecto:
    mkdir servidor-websocket
cd servidor-websocket
    1. Inicializa un proyecto de Node:
    npm init -y
    1. Instala la librería ws (WebSocket):
    npm install ws

    3. ✅ Crear el servidor WebSocket básico#

    1. Crea un archivo llamado index.js:
    touch index.js
    1. Copia el siguiente código en index.js:
    const WebSocket = require('ws');

const wss = new WebSocket.Server({ port: 8080 });

wss.on('connection', (ws) => {
  console.log('Nuevo cliente conectado');

  ws.on('message', (message) => {
    console.log(`Mensaje recibido: ${message}`);

    // Reenviar el mensaje a todos los clientes conectados
    wss.clients.forEach(client => {
      if (client !== ws && client.readyState === WebSocket.OPEN) {
        client.send(message);
      }
    });
  });

  ws.on('close', () => {
    console.log('Cliente desconectado');
  });
});

console.log('Servidor WebSocket escuchando en ws://localhost:8080');
    1. Ejecuta el servidor:
    node index.js

    Verás el mensaje: Servidor WebSocket escuchando en ws://localhost:8080

    4. 🧪 Probar el servidor (opcional)#

    Puedes usar una herramienta como:

    • WebSocket Echo Test para conectarte a tu servidor local.
    • O crear un pequeño cliente en Unity con NativeWebSocket.

    5. 📄 Archivos recomendados:#

    • index.js → contiene el código del servidor.
    • package.json → generado con npm init, gestiona dependencias.
    • node_modules/ → carpeta automática con las librerías instaladas.

    ℹ️ Notas:#

    • El puerto 8080 debe estar libre en tu sistema.
    • El servidor no tiene HTTPS por defecto.
    • Puedes personalizarlo para trabajar con salas, autenticación, etc.

    Código soportar salas#

    const WebSocket = require('ws');
const wss = new WebSocket.Server({ port: 8080 });

const clients = new Map(); // ws -> { username, room }
const rooms = {}; // roomName -> Set of ws

function broadcast(room, message, sender) {
  if (!rooms[room]) return;
  rooms[room].forEach(client => {
    if (client !== sender && client.readyState === WebSocket.OPEN) {
      client.send(message);
    }
  });
}

wss.on('connection', (ws) => {
  clients.set(ws, { username: null, room: null });
  ws.send('Bienvenido! Usa /join [sala] para entrar a una sala.');

  ws.on('message', (data) => {
    const msg = data.toString().trim();
    const clientData = clients.get(ws);

    if (msg.startsWith('/join')) {
      const roomName = msg.split(' ')[1];
      if (!roomName) return ws.send('Uso: /join nombre_sala');

      if (clientData.room) {
        rooms[clientData.room]?.delete(ws);
      }

      clientData.room = roomName;
      if (!rooms[roomName]) rooms[roomName] = new Set();
      rooms[roomName].add(ws);

      ws.send(`Te has unido a la sala '${roomName}'.`);
      return;
    }

    if (!clientData.room) {
      return ws.send('Debes unirte a una sala con /join [nombre] antes de enviar mensajes.');
    }

    broadcast(clientData.room, msg, ws);
  });

  ws.on('close', () => {
    const clientData = clients.get(ws);
    if (clientData?.room) rooms[clientData.room]?.delete(ws);
    clients.delete(ws);
  });
});

console.log('Servidor con soporte de salas activo en ws://localhost:8080');
